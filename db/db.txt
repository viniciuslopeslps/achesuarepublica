drop database acheSuaRepublica;
create database acheSuaRepublica;
use acheSuaRepublica;

create table users
(
id_usu int primary key auto_increment,
name_usu varchar(50) not null,
email_usu varchar(40) not null unique,
password_usu varchar(100) not null,
phone_user varchar(15) not null,
admin_user bool not null
);

create table location
(
id_locat int primary key auto_increment,
city_locat varchar(40) not null unique,
state_locat varchar(2) not null,
address_locat varchar(60) not null,
key_locat varchar(60) not null,
number_locat int,
id_usu int references users(id_usu)
);
